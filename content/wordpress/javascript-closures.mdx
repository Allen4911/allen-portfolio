Closures are one of the most frequently misunderstood concepts in JavaScript. They're a staple of interview questions, but more importantly, they're patterns you use every day. This post focuses on how closures are actually used in real-world code, beyond just theory.

## What is a Closure?

A closure is **a function that remembers the lexical scope in which it was declared**. Even when the function is executed outside of that scope, it can still access that scope.

```javascript
function makeCounter() {
  let count = 0; // local variable of makeCounter
  
  return function increment() {
    count++; // accessing outer variable — closure!
    return count;
  };
}

const counter = makeCounter(); // makeCounter execution completes
// At this point, count should be gone...

console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3
// increment keeps referencing count → count stays alive
```

Even after `makeCounter` returns, the `count` variable persists in memory. This is because the returned `increment` function holds a reference to `count`.

## Lexical Scope

JavaScript uses **Lexical Scope**. The scope is determined by where the function is **declared**, not where it is called.

```javascript
let x = 'global';

function outer() {
  let x = 'outer';
  
  function inner() {
    console.log(x); // 'outer' — scope of declaration site
  }
  
  return inner;
}

const fn = outer();
fn(); // 'outer' (not the global x at the call site)
```

## Practical Closure Patterns

### 1. Data Hiding (Private State)

JavaScript classes didn't have true private fields (before ES2022 `#field`). Closures provided the solution:

```javascript
function createAccount(initialBalance) {
  let balance = initialBalance; // private — not directly accessible from outside
  const transactions = [];
  
  return {
    deposit(amount) {
      if (amount &lt;= 0) throw new Error('Amount must be positive');
      balance += amount;
      transactions.push({ type: 'deposit', amount, balance });
      return balance;
    },
    
    withdraw(amount) {
      if (amount > balance) throw new Error('Insufficient balance');
      balance -= amount;
      transactions.push({ type: 'withdraw', amount, balance });
      return balance;
    },
    
    getBalance() { return balance; },
    getHistory() { return [...transactions]; } // returns a copy
  };
}

const account = createAccount(1000);
account.deposit(500);  // 1500
account.withdraw(200); // 1300
console.log(account.balance); // undefined — direct access not possible!
```

### 2. Function Factories

```javascript
// Multiplication function factory
function multiplier(factor) {
  return (number) => number * factor;
}

const double = multiplier(2);
const triple = multiplier(3);
const tenX   = multiplier(10);

console.log(double(5));  // 10
console.log(triple(5));  // 15
console.log(tenX(5));    // 50

// Practical example: permission check factory
function requireRole(role) {
  return function(handler) {
    return function(req, res) {
      if (req.user?.role !== role) {
        return res.status(403).json({ error: 'Forbidden' });
      }
      return handler(req, res);
    };
  };
}

const requireAdmin = requireRole('admin');
app.delete('/users/:id', requireAdmin(deleteUserHandler));
```

### 3. Memoization

```javascript
function memoize(fn) {
  const cache = new Map(); // cache preserved via closure
  
  return function(...args) {
    const key = JSON.stringify(args);
    
    if (cache.has(key)) {
      console.log('Cache hit:', key);
      return cache.get(key);
    }
    
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

const expensiveCalculation = memoize(function(n) {
  console.log(`Computing: ${n}`);
  return n * n * n;
});

expensiveCalculation(5); // "Computing: 5" → 125
expensiveCalculation(5); // "Cache hit" → 125 (no recomputation)
expensiveCalculation(3); // "Computing: 3" → 27
```

### 4. Partial Application

```javascript
function partial(fn, ...presetArgs) {
  return function(...laterArgs) {
    return fn(...presetArgs, ...laterArgs);
  };
}

function add(a, b, c) {
  return a + b + c;
}

const add5 = partial(add, 5);       // a = 5 fixed
const add5and3 = partial(add, 5, 3); // a = 5, b = 3 fixed

console.log(add5(3, 2));    // 10
console.log(add5and3(7));   // 15

// Practical example: setting API call defaults
const apiCall = partial(fetch, 'https://api.example.com');
const getUser = partial(apiCall, '/users');
```

### 5. Event Handlers and Closures

```javascript
// Beware: closure trap with var
for (var i = 0; i &lt; 3; i++) {
  setTimeout(() => console.log(i), 100); // 3, 3, 3 — expected: 0, 1, 2
}

// Fix 1: use let (block scope)
for (let i = 0; i &lt; 3; i++) {
  setTimeout(() => console.log(i), 100); // 0, 1, 2 ✓
}

// Fix 2: capture with closure
for (var i = 0; i &lt; 3; i++) {
  ((j) => {
    setTimeout(() => console.log(j), 100); // 0, 1, 2 ✓
  })(i);
}
```

## Closures and Memory

Since closures reference outer scope variables, those variables are not garbage collected.

```javascript
function createHeavyObject() {
  const largeArray = new Array(1000000).fill('data'); // large array
  
  return function() {
    return largeArray.length; // keeps referencing largeArray
  };
}

const fn = createHeavyObject();
// largeArray stays in memory as long as fn exists

// To release it:
fn = null; // remove reference → largeArray can be garbage collected
```

In Node.js servers, closure leaks can cause serious memory problems. Explicitly remove references to closures you no longer need.

## Closures vs Classes

Comparing the same functionality with closures and classes:

```javascript
// Closure approach
function createUser(name, role) {
  return {
    getName: () => name,
    isAdmin: () => role === 'admin',
    greet: () => `Hello, ${name}`
  };
}

// Class approach
class User {
  #name;
  #role;
  
  constructor(name, role) {
    this.#name = name;
    this.#role = role;
  }
  
  getName() { return this.#name; }
  isAdmin() { return this.#role === 'admin'; }
  greet() { return `Hello, ${this.#name}`; }
}
```

The closure approach uses more memory due to no prototype chain sharing, but avoids `this` binding issues. It aligns well with functional programming style.

## Practical Checklist

- `var` + loop with closures → replace with `let` or IIFE
- When closures reference large data → extract and store only the needed values
- Memoization cache size → prevent unbounded growth (consider LRU cache pattern)
- Implementing private state with closures → also consider ES2022 `#field` as an alternative

Closures are at the heart of JavaScript's functional characteristics. React's `useState`, Redux middleware, and Express middleware chains all operate on top of closures. Simple in concept, but deeply understanding them enables safer and more expressive code.
